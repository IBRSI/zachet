// Импортируйте необходимые модули import * as THREE from 'three'; import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader.js'; // Создайте сцену, камеру и рендерер const scene = new THREE.Scene(); const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000); const renderer = new THREE.WebGLRenderer(); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); // Загрузите модель кролика const rabbitLoader = new GLTFLoader(); rabbitLoader.load('path/to/rabbit.glb', (gltf) => { const rabbit = gltf.scene; scene.add(rabbit); }); // Функция для превращения кролика в шляпу function transformRabbitToHat() { // Удалите кролика из сцены scene.remove(rabbit); // Загрузите модель шляпы const hatLoader = new GLTFLoader(); hatLoader.load('path/to/hat.glb', (gltf) => { const hat = gltf.scene; scene.add(hat); }); } // Пример вызова функции document.addEventListener('keydown', (event) => { if (event.key === 't') { // Например, при нажатии на 't' transformRabbitToHat(); } }); // Настройте камеру camera.position.z = 5; // Анимация function animate() { requestAnimationFrame(animate); renderer.render(scene, camera); } animate();
